3D Модели и Google Model Viewer

Google Model Viewer

Основной инструмент для отображения 3D моделей в проекте Beresta.

Настройки Model Viewer

<model-viewer 
  src="/models/beresta.glb" 
  alt="3D модель Бересты" 
  camera-controls 
  auto-rotate 
  ar 
  fallback-src="/img/device-3d.png"
  style="width:100%; height:500px; background:#111; border-radius:10px;">
</model-viewer>

Поддерживаемые форматы

📦
GLB
Binary GLTF

Основной формат для 3D моделей. Содержит геометрию, материалы и анимации.

📁
GLTF
JSON + Binaries

Текстовый формат GLTF с отдельными файлами для текстур и геометрии.

🎨
FBX
Autodesk FBX

Формат для экспорта из 3D редакторов. Конвертируется в GLB.

Атрибуты Model Viewer

// Основные атрибуты
camera-controls     // Управление камерой мышью
auto-rotate         // Автоматическое вращение
ar                  // Поддержка AR
fallback-src        // Изображение-заглушка
interaction-policy  // Политика взаимодействия
camera-orbit        // Начальная позиция камеры
field-of-view       // Угол обзора
min-camera-orbit    // Ограничения камеры
max-camera-orbit    // Ограничения камеры

// Стилизация
style="width:100%; height:500px; background:#111; border-radius:10px;"

JavaScript API

// Получение элемента
const modelViewer = document.querySelector('model-viewer');

// События
modelViewer.addEventListener('load', () => {
  console.log('Модель загружена');
});

modelViewer.addEventListener('error', (event) => {
  console.error('Ошибка загрузки:', event.detail);
});

// Управление анимацией
modelViewer.play();
modelViewer.pause();

// Управление камерой
modelViewer.cameraOrbit = '0deg 75deg 1.5m';
modelViewer.fieldOfView = '30deg';

JavaScript принципы

// 1. Используйте современный ES6+ синтаксис
class MatrixController {
  constructor() {
    this.pixels = [];
    this.isEnabled = false;
  }

  // 2. Асинхронные операции
  async connect() {
    try {
      await this.bluetooth.connect();
      this.updateStatus('connected');
    } catch (error) {
      console.error('Connection failed:', error);
    }
  }

  // 3. Обработка событий
  handlePixelClick(event) {
    const pixel = event.target;
    pixel.classList.toggle('on');
    this.updateMatrix();
  }
}
